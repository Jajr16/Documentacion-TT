\section{IU17 Pantalla de Reconocimiento facial}

\subsection{Objetivo}
Permitir al alumno registrar su asistencia al ETS.

\subsection{Diseño}
Esta pantalla \IUref{IU17}{Pantalla de Reconocimiento facial} (ver figura~\ref{IU17}) aparece luego de seleccionar el botón \IUbutton{Registrar asistencia}.

\IUfig[.35]{cu19}{IU17}{Pantalla de Reconocimiento facial}

\subsection{Salidas}
Confirmación de asistencia registrada

\subsection{Entradas}
Ninguna


\subsection{Comandos}
\begin{itemize}
	\item \IUbutton{Cancelar}: Permite al alumno cancelar la operación de Reconocimeinto facial.
	\item \IUbutton{Comenzar}: Activa la cámara para el Reconocimiento facil. 
\end{itemize}

\subsection{Mensajes}

\begin{Citemize}
	\item {\bf MSG-22-}{``No se pudo activar la cámara o reconocer la identidad. Intente nuevamente.''}
\end{Citemize}
